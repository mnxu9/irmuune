// ── HEART ──
function initHeart(){
  const canvas=document.getElementById('hcanvas');
  const ctx=canvas.getContext('2d');
  const DPR=Math.min(window.devicePixelRatio||1,2);
  const availW=window.innerWidth;
  const availH=window.innerHeight-60;
  const SIZE=Math.min(availW,availH,480);
  canvas.style.width=SIZE+'px';
  canvas.style.height=SIZE+'px';
  canvas.width=Math.round(SIZE*DPR);
  canvas.height=Math.round(SIZE*DPR);
  ctx.scale(DPR,DPR);
  const W=SIZE,H=SIZE;

  const NAMES=["irmuune ","irmuunzul ","i love you "];
  let nameIdx=0,TEXT=NAMES[0];
  setInterval(()=>{nameIdx=(nameIdx+1)%NAMES.length;TEXT=NAMES[nameIdx];},5000);

  const FONT=13;
  const ROWS=15;
  const ROW_GAP=0.50;
  const STEPS=400;
  const CHAR_SP=FONT*0.88;

  function hx(t){return 16*Math.pow(Math.sin(t),3);}
  function hy(t){return-(13*Math.cos(t)-5*Math.cos(2*t)-2*Math.cos(3*t)-Math.cos(4*t));}

  const bs=Math.min(W,H)/38;
  const cx=W/2,cy=H/2+6;

  const lanes=[];
  for(let r=0;r<ROWS;r++){
    const s=bs-r*ROW_GAP;
    if(s<=1)break;
    const pts=[];
    for(let i=0;i<=STEPS;i++){
      const t=i/STEPS*Math.PI*2;
      pts.push({x:cx+hx(t)*s,y:cy+hy(t)*s});
    }
    const cumLen=[0];
    for(let i=0;i<pts.length-1;i++){
      const dx=pts[i+1].x-pts[i].x,dy=pts[i+1].y-pts[i].y;
      cumLen.push(cumLen[i]+Math.sqrt(dx*dx+dy*dy));
    }
    const totalLen=cumLen[cumLen.length-1];
    const chars=[];
    let ci=0;
    for(let dist=0;dist<totalLen-CHAR_SP;dist+=CHAR_SP){
      let lo=0,hi=cumLen.length-1;
      while(lo<hi-1){const m=(lo+hi)>>1;if(cumLen[m]<=dist)lo=m;else hi=m;}
      const f=(dist-cumLen[lo])/(cumLen[hi]-cumLen[lo]||1);
      const p0=pts[lo],p1=pts[hi];
      chars.push({
        x:p0.x+(p1.x-p0.x)*f,
        y:p0.y+(p1.y-p0.y)*f,
        angle:Math.atan2(p1.y-p0.y,p1.x-p0.x),
        bi:ci++
      });
    }
    lanes.push({chars,r});
  }

  const off=document.createElement('canvas');
  off.width=W;off.height=H;
  const octx=off.getContext('2d');
  octx.textAlign='center';
  octx.textBaseline='middle';

  let t=0;
  let rafId;
  let lastTime=0;
  const FRAME_MS=1000/40;

  function draw(now){
    rafId=requestAnimationFrame(draw);
    if(now-lastTime<FRAME_MS)return;
    lastTime=now;
    octx.clearRect(0,0,W,H);
    octx.font=`bold ${FONT}px 'Courier New',monospace`;
    lanes.forEach(({chars,r})=>{
      const speed=0.15+r*0.010;
      const la=1-(r/ROWS)*0.40;
      const rv=Math.max(190,234-r*5);
      const gv=Math.max(72,128-r*7);
      const bv=Math.min(202,170+r*3);
      const shift=Math.floor(t*speed);
      chars.forEach((c,i)=>{
        const ch=TEXT[(c.bi+shift)%TEXT.length];
        const wave=0.70+0.30*Math.sin(t*0.032+i*0.18+r*0.38);
        octx.fillStyle=`rgba(${rv},${gv},${bv},${wave*la})`;
        octx.save();
        octx.translate(c.x,c.y);
        octx.rotate(c.angle);
        octx.fillText(ch,0,0);
        octx.restore();
      });
    });
    ctx.clearRect(0,0,W,H);
    ctx.save();ctx.filter='blur(7px)';ctx.globalAlpha=0.28;ctx.drawImage(off,0,0);ctx.restore();
    ctx.save();ctx.filter='blur(2.5px)';ctx.globalAlpha=0.20;ctx.drawImage(off,0,0);ctx.restore();
    ctx.save();ctx.globalAlpha=1;ctx.drawImage(off,0,0);ctx.restore();
    t++;
  }

  rafId=requestAnimationFrame(draw);
  document.addEventListener('visibilitychange',()=>{
    if(document.hidden)cancelAnimationFrame(rafId);
